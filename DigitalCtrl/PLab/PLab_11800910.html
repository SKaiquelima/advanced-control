T_s = T # Perodo de amostragem 
# para plotar por masi de 1.5, tem que aumentar esse Ts, por isso, dividi em duas vriaveis
# a prnio, com t at 1.5, Ts = T fica visualmente bom no plot

t = np.arange(0, 1.5 + T_s, T_s) 

# Inicializar variveis
u = np.zeros_like(t)  # Sada do controlador
e = np.ones_like(t)   # Entrada de erro (degrau unitrio)
I = 0                
previous_error = 0   


for n in range(len(t)):
    # Erro atual (e[n]  sempre 1 neste caso)
    error = e[n]
    
    # Termo proporcional
    P = Kp * error
    
    # Termo integral (aproximao de Euler Forward)
    I = I + error * T_s  # Acumular o integral
    I_term = Ki * I
    
    # Termo derivativo (aproximao de Euler Forward)
    if n == 0:
        D = 0  # Derivada  zero na primeira amostra
    else:
        D = Kd * (error - previous_error) / T_s
    
    # Calcular a sada do controlador
    u[n] = P + I_term + D
    
    previous_error = error
